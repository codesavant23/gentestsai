```python
from typing import List, Dict, Any


def calculate_discount(price: float, age: int, is_member: bool) -> float:
	if price <= 0:
		raise ValueError("The price cannot be negative or zero")
	if age <= 0:
		raise ValueError("The age of the buyer cannot be negative or zero")

	discount = 0.0

	if age < 12:
		discount += 0.5
	elif age >= 65:
		discount += 0.3

	if is_member:
		discount += 0.1

	final_price = price * (1 - discount)
	return max(final_price, 0.0)


def calculate_cart_total(items: List[Dict[str, Any]], tax_rate: float = 0.22, shipping: float = 5.0, free_shipping_threshold: float = 50.0) -> float:
	if (items is None) or (len(items) == 0):
		return 0.0

	subtotal: float = 0.0
	for item in items:
		price = item.get("price", 0.0)
		qty = item.get("quantity", 1)
		discount = item.get("discount", 0.0)

		if (price < 0) or (qty < 0) or (not (0.0 <= discount <= 1.0)):
			raise ValueError("Invalid data for a cart item")

		subtotal += price * qty * (1 - discount)

	taxed_total = subtotal * (1 + tax_rate)

	if subtotal >= free_shipping_threshold:
		return round(taxed_total, 2)
	else:
		return round(taxed_total + shipping, 2)
```

```python
def test_no_cart():
    # Empty cart -> Total = 0
    assert calculate_cart_total([]) == 0.0

    # Invalid cart -> Total = 0
    assert calculate_cart_total(None) == 0.0


def test_invalid_price():
    # Invalid carts (with invalid price) -> ValueError Exception
    cart1_inv_price = [{"price": -3.7, "quantity": 1}]
    cart2_inv_price = [{"price": 0, "quantity": 1}]

    with pytest.raises(ValueError):
        calculate_cart_total(cart1_inv_price)
    with pytest.raises(ValueError):
        calculate_cart_total(cart2_inv_price)
```
