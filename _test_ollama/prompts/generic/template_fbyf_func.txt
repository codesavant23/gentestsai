from typing import List, Dict, Any


def calculate_discount(price: float, age: int, is_member: bool) -> float:
	"""
	Calculates the discounted price based on age and loyalty program membership.

	Rules:
		- If `price` is negative, or zero, a ValueError exception is raised.
		- If `age` is negative, or zero, a ValueError exception is raised.
		- Children (<12 years) receive a 50% discount.
		- Seniors (>=65 years) receive a 30% discount.
		- Members receive an additional 10% discount on the total.

	Args:
		price: float
			Original price of the item.
		age: int
			Age of the buyer.
		is_member: bool
			True if the buyer is a loyalty program member.

	Returns:
		float
			Final price after discounts.
	"""

	if price <= 0:
		raise ValueError("The price cannot be negative or zero")
	if age <= 0:
		raise ValueError("The age of the buyer cannot be negative or zero")

	discount = 0.0

	if age < 12:
		discount += 0.5
	elif age >= 65:
		discount += 0.3

	if is_member:
		discount += 0.1

	final_price = price * (1 - discount)
	return max(final_price, 0.0)


def calculate_cart_total(items: List[Dict[str, Any]], tax_rate: float = 0.22, shipping: float = 5.0, free_shipping_threshold: float = 50.0) -> float:
	"""
	Calculate a cart total, applying taxes and shipping.

	Rules:
	    - If `items` is an empty list or is None then the final cart total is 0
	    - If any of the cart items is not valid, a ValueError exception is raised

	Args:
		items: List[Dict[str, Any]]
			List of the cart items.
			Each valid cart item is a dictionary with:
			    - "price": float,
			    - "quantity": int,
			    - (optional) "discount": float (from 0.0 to 1.0)
		tax_rate: float
			VAT percentage to apply to the subtotal. (Default is 0.22)
		shipping: float
			Shipping cost to add.
		free_shipping_threshold: float
			Threshold above which shipping is free.

	Returns:
		float
		    Final cart total.
	"""
	if (items is None) or (len(items) == 0):
		return 0.0

	subtotal: float = 0.0
	for item in items:
		price = item.get("price", 0.0)
		qty = item.get("quantity", 1)
		discount = item.get("discount", 0.0)

		if (price < 0) or (qty < 0) or (not (0.0 <= discount <= 1.0)):
			raise ValueError("Invalid data for a cart item")

		subtotal += price * qty * (1 - discount)

	taxed_total = subtotal * (1 + tax_rate)

	if subtotal >= free_shipping_threshold:
		return round(taxed_total, 2)
	else:
		return round(taxed_total + shipping, 2)
