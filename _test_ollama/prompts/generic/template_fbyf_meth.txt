from typing import List, Dict, Any


class ShoppingCart:
	def __init__(self, tax_rate: float = 0.22, shipping: float = 5.0, free_shipping_threshold: float = 50.0):
		"""
		Creates, and initializes, a new shopping cart.

		Args:
			tax_rate: float
				VAT percentage to apply to the subtotal (Default is 0.22).
			shipping: float
				Shipping cost to add.
			free_shipping_threshold: float
				Threshold above which shipping is free.
		"""
		self._items: List[Dict[str, Any]] = list()
		self._tax_rate: float = tax_rate
		self._ship_fee: float = shipping
		self._free_ship_threshold: float = free_shipping_threshold


	def add_item(self, name: str, price: float, quantity: int = 1, discount: float = 0.0):
		"""
		Adds a new product to the shopping cart.

		Args:
			name: str
				Name of the product.
			price: float
				Unitary price of the product.
			quantity: int
				Product quantity (Default is 1).
			discount: float
				Discount percentage (Default is 0.0)
		"""
		if (price < 0) or (quantity <= 0) or (not (0.0 <= discount <= 1.0)):
			raise ValueError("Invalid data for the product given")

		self._items.append({
			"name": name,
			"price": price,
			"quantity": quantity,
			"discount": discount
		})


	def remove_item(self, name: str):
		"""
		Removes an existing product from the shopping cart.

		Rules:
			- If `name` contains the name of a product that doesn't exist, a ValueError exception is raised

		Args:
			name: str
				Name of the product to remove from the shopping cart.
		"""
		i: int = 0
		prod_index: int = -1
		prod_exists: bool = False
		while (not prod_exists) and (i < len(self._items)):
			if item["name"] == name:
				prod_exists = True
				prod_index = i

		if not prod_exists:
			raise ValueError("The product with name '" + name + "' doesn't exist")

		del self._items[prod_index]


	def calculate_total(self) -> float:
		"""
		Calculate the final price of the entire cart.

		Returns:
			float
				Final cart total.
		"""
		if (len(self._items) == 0):
			return 0.0

		subtotal: float = 0.0
		for item in self._items:
			subtotal += item["price"] * item["quantity"] * (1 - item["discount"])

		taxed_total = subtotal * (1 + self.tax_rate)

		actual_ship: float = self._ship_fee
		if subtotal >= self._free_ship_threshold:
			actual_ship = 0.0

		return round(taxed_total + actual_ship, 2)
